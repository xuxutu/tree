4.不用define的情况下，定义的时候变量名对数组设定大小
  4.1 对一个const变量取地址合法，对一个enum取地址合法
class GamePlayer
{
public:
    void show()
    {
        cout<<sizeof(scores)/sizeof(int)<<endl;
        cout<<NumTurns<<endl;
        //cout<<&NumTurns<<endl;
    }

private:
    enum{NumTurns=5};
    int scores[NumTurns];
};

int main(void)
{
    GamePlayer gp;
    gp.show();

    return 0;
}
