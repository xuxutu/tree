1.格式：

   while（表达式）
      
         语句；

#include<stdio.h>
#include<stdlib.h>

int main(void)
{
    int i=0;                           //i计数
    int row=1;                         //定义行

	while(i<1000)
	{
		
		if(row%2!=0)                   //奇数行成立时，执行
		{
			int j=0;

	        while(j<5)                 //j用来判断当前行已经输出的个数，当输出5个值时，跳出循环；
			{
		        i++;                   //i加1

                if(i>1000)
					break;

			    if(i%10==1&&i%3==0)    //符合条件时（尾数是1且能被3整除）执行
				{
		            printf("%5d",i);   //输出符合条件的i值
					j++;               //每输出一个i值（尾数是1且能被3整除时输出），才使j加1；
				}

			}

            row++;                     //行计数加1
			printf("\n");              //换行
		}
		else                           //偶数行成立时执行
		{
			int k=0;

		    while(k<10)                //k用来判断当前行已经输出的个数，当输出10个值时，跳出循环；
			{
		        i++; //i加1

                if(i>1000)
					break;

			    if(i%10==1&&i%3==0)    //符合条件时（尾数是1且能被3整除）执行
				{
		             printf("%5d",i);  //输出符合条件的数
					 k++;              //每输出一个i值（尾数是1且能被3整除时输出），才使k加1；
				}

			}

			row++;                     //行计数加一
			printf("\n");              //换行
		}
		
	}   
	
    system("pause");

}
